Here's how I implemented it:

char **ft_split(char const *s, char c)
{
    // Count words
    int word_count = 0;
    int i = 0;
    
    while (s[i])
    {
        if (s[i] != c && (s[i + 1] == c || s[i + 1] == '\0'))
            word_count++;
        i++;
    }
    
    // Allocate memory
    char **result = (char **)malloc(sizeof(char *) * (word_count + 1));
    if (!result)
        return (NULL);
        
    // Split logic here...
    
    return (result);
}

The key is to first count the words, then allocate memory for each word.
